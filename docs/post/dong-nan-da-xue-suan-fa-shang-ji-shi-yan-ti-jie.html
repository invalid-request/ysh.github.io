1007 最大乘积

题目

设有一个长度为N的数字串，要求选手使用K个乘号将它分成K+1个部分，找出一种分法，使得这K+1个部分的乘积能够为最大。
同时，为了帮助选手能够正确理解题意，主持人还举了如下的一个例子：
有一个数字串：312， 当N=3，K=1时会有以下两种分法：

  1. 3*12=36
  2. 31*2=62

这时，符合题目要求的结果是：31*2=62
现在，请你帮助你的好朋友XZ设计一个程序，求得正确的答案。

思路

  1. 采用string处理长整数，自定义乘法、大于、截取运算。
  2. dp[i][j]：前i个数字，插入j个乘号的最大乘积

关键代码

// 数字长度为n，k个乘号
string solve(int n, int k, const string& s) {
    // dp[i][j]：前i个数字，插入j个乘号的最大乘积
    vector<vector<string>> dp(n + 1, vector<string>(k + 1, "0"));
    
    for (int i = 1; i <= n; ++i) {
        dp[i][0] = getNum(s, 0, i - 1);
    }
    
    for (int j = 1; j <= k; ++j) { 
        for (int i = j + 1; i <= n; ++i) { 
            for (int m = j; m < i; ++m) {
                // 遍历最右边乘号的位置
                string t = dp[m][j - 1]*getNum(s, m, i - 1);
                if (isGreaterThan(t,dp[i][j])) {
                    dp[i][j] = t;
                }
            }
        }
    }
    return dp[n][k];
}

1008 拦截导弹2

题目

某国为了防御敌国的导弹袭击，发展出一种导弹拦截系统。但是这种导弹拦截系统有一个缺陷：虽然它的第一发炮弹能够到达任意的高度，但是以后每一发炮弹都不能高于前一发的高度。某天，雷达捕捉到敌国的导弹来袭。由于该系统还在试用阶段，所以只有一套系统，因此有可能不能拦截所有的导弹。

第一行输入M表示包含M组测试数据，每组第一个输入N (N<100)表示后面有N个整数，表示导弹依次飞来的高度(雷达给出的高度数据是不大于30000的正整数)。

对于每组输入数据，第一行输出这套系统最多能拦截多少导弹，以及输出如果要拦截所有导弹最少要配备多少套这种导弹拦截系统。

思路

  1. 一套系统最多能拦截多少导弹 = 原导弹高度序列的最长不上升子序列长度。
  2. 最少需要多少系统 = 原序列的最长上升子序列长度。
  3. MaxNum中**t[i]:以下标i结尾的最长不上升子序列长度**。
  4. MinNum中**t[i]:以下标i结尾的最长上升子序列长度**。

关键代码

// 求导弹高度序列的最长不上升子序列
int MaxNum(vector<int>& a) {
    // t[i]:以下标i结尾的最长不上升子序列长度
    vector<int> t(a.size());
    t[0]=1;
    int max=1;
    for(int i=1;i<a.size();i++) {
        t[i]=1;
        for(int j=0;j<i;j++) {
            if(a[j]>=a[i]) {
                t[i]=t[i]>=(t[j]+1)?t[i]:(t[j]+1);
            }
        }
        max=max>=t[i]?max:t[i];
    }
    return max;
}

// 求导弹高度序列的最长上升子序列长度
int MinNum(vector<int>& a) {
    // t[i]:以下标i结尾的最长上升子序列长度
    vector<int> t(a.size());
    t[0]=1;
    int min=1;
    for(int i=1;i<a.size();i++) {
        t[i]=1;
        for(int j=0;j<i;j++) {
            if(a[j]<a[i]) {
                t[i]=t[i]>=(t[j]+1)?t[i]:(t[j]+1);
            }
        }
        min=min>=t[i]?min:t[i];
    }
    return min;
}

题目

给定一递增有序数组a[0,1,...,n-1], 请在数组中搜索给定元素。搜索过程中请使用mid=(low+high)/2。

搜索成功输出success及父亲, 否则输出not found及父亲。

思路

  1. 二分搜索。
  2. 每次循环若未命中。

关键代码

int low = 0, high = N - 1;
int father = 0;    // 记录最终的父节点
bool find = false;	// 记录是否找到

while (low <= high)
{
    // 题目指定 mid = (low + high) / 2
    int mid = (low + high) / 2;
    if (a[mid] == X)
    {
        find = true;
        break;
    }
    else if (a[mid] < X)
    {
        low = mid + 1;
    }
    else
    {
        high = mid - 1;
    }
    father = a[mid]; // 每轮更新父节点为当前mid值
}

1013 逆序对

题目

给定一个长度为N的int型数组a[0,1,2,...N-1], 请计算逆序对个数。当i<j且a[i]>a[j], 则称a[i]与a[j]是一对逆序对。

思路

  1. 在归并排序中进行逆序对的计数

关键代码

// 用一维数组t临时存储归并过程中合并两序列的结果
// 事先创建好，令其长度等于需要计算逆序对个数的数组的长度
vector<int> t;

// 对数组a下标l到r的元素进行归并排序
int merge(vector<int>& a, int l, int r) {
    int res=0;	// 存储这一次归并中逆序对的个数
    if(l>=r) return res;
    
    // 先对左边一半和右边一半进行排序
    // 并将两次排序发现的逆序对个数记录进res中
    int mid=(l+r)/2;
    res+=merge(a,l,mid);
    res+=merge(a,mid+1,r);
    
    int p=l;		// 左指针
    int q=mid+1;	// 右指针
    int x=l;		// 一维临时数组t的指针
    while(p<=mid&&q<=r) {
        if(a[p]<=a[q]) {
            t[x++]=a[p];
            p++;
        }
        else {
            t[x++]=a[q];
            res+=mid-p+1;	// 发现mid-p+1个逆序对
            q++;
        }
    }
    while(p<=mid) {
        t[x++]=a[p++];
    }
    while(q<=r) {
        t[x++]=a[q++];
    }
    for(int i=l;i<=r;i++) {
        a[i]=t[i];
    }
    return res;
}

1014 最大子数组和

题目

给定一个长度为N的int型数组a[0,1,2,...N-1], 请计算最大子数组和。

思路

  1. x[i]：以下标i的元素结尾的最大子数组和。
  2. 再遍历一遍数组x找出最大值。

关键代码

// 找出数组a的最大子数组和
int f(vector<int>& a) {
    // x[i]：以下标 i 的元素结尾的最大子数组和
    vector<int> x(a.size());
    x[0]=a[0];
    for(int i=1;i<a.size();i++) {
        if(x[i-1]>0) x[i]=a[i]+x[i-1];
        else x[i]=a[i];
    }
    
    int max=x[0];
    for(int i=1;i<a.size();i++) {
        max=max>=x[i]?max:x[i];
    }
    return max;
}

1016 两元素和

题目

给定一个N(N<=50000)个int型整数的集合以及一个int型整数X, 问集合中是否存在两个元素的和等于X。

思路

  1. 使用哈希表存储访问过的元素。
  2. 每次访问时寻找(X - 当前元素)是否在哈希表中。

关键代码

// 判断数组a中是否存在两个元素之和为t
bool f(vector<int>& a, int t) {
    unordered_set<int> x;	// 哈希表x存储访问过的元素
    x.insert(a[0]);
    for(int i=1;i<a.size();i++) {
        if(x.count(t-a[i])) return 1;
        x.insert(a[i]);
    }
    return 0;
}

1018 0/1背包问题2

题目

有一个容量为C(C<=100)的奇怪背包，这个背包可以被带走仅当它恰好被装满。现在你手边有N(N<=500)颗宝石，第i颗宝石大小为si，价值为vi。由于条件限制，你手边只有这个奇怪的背包可作为你搬运宝石的唯一工具。现在你想知道在这样的条件下你最多可以带走多大利润的宝石。

思路

  1. 自定义结构体gem存储一个宝石的重量与价值。
  2. dp[i]：装总重量为 i 的宝石的最大价值。
  3. 遍历所有宝石，每个宝石分别从dp[capacity]开始倒着遍历，对于dp[j]，只有dp[j-w]>0时才说明存在把背包装到总重量j的方案。
  4. 最后返回dp[capacity]。

关键代码

// 自定义宝石结构体用来存储重量w与价值v
struct gem {
    int w, v;
};

// 面对a这堆宝石，在刚好装满容量为capacity的背包的前提下，能带走的最大价值
int knapsack(vector<gem>& a, int capacity) {
    int n = a.size();
    // 装总重量为i的宝石的最大价值
    vector<int> dp(capacity + 1, 0);
    
    for (int i = 0; i < n; i++) {
        int w = a[i].w;
        int v = a[i].v;
        for (int j = capacity; j >= w; j--) {
            if(j-w>0) {
                // dp[j-w]>0的话才说明存在把背包装到总重量j的方案
                dp[j]=(dp[j-w]>0)?(max(dp[j], dp[j - w] + v)):dp[j];
            }
            else {
                dp[j] = max(dp[j], v);
            }
        }
    }
    return dp[capacity];
}

1019 0/1背包问题1

题目

有一个容量为C(C<=100)的背包以及N(N<=500)颗宝石，第i颗宝石大小为si，价值为vi。由于条件限制，你手边只有这个背包可作为你搬运宝石的唯一工具。现在你想知道在最多可以带走多大价值的宝石。

思路

  1. 自定义结构体gem存储一个宝石的重量与价值。
  2. dp[i]：装总重量为 i 的宝石的最大价值。
  3. 遍历所有宝石，每个宝石分别从dp[capacity]开始倒着遍历，dp[j]等于dp[j-w]+v与dp[j]的最大值。
  4. 最后返回dp[capacity]。

关键代码

// 自定义宝石结构体用来存储重量w与价值v
struct gem {
    int w, v;
};

// 面对a这堆宝石，在背包容量为capacity的前提下，能带走的最大价值
int knapsack(vector<gem>& a, int capacity) {
    int n = a.size();

    for (int i = 0; i < n; i++) {
        int w = a[i].w;
        int v = a[i].v;
        for (int j = capacity; j >= w; j--) {
            dp[j] = max(dp[j], dp[j - w] + v);
        }
    }
    return dp[capacity];
}

1020 矩阵连乘

题目

两个矩阵A(r行s列)和B(s行t列)相乘, 乘法代价为r*s*t. 现给定N(N<=500)个矩阵连乘问题, 请计算最小乘法代价。

思路

  1. dp[i][j]：从下标为i的矩阵连乘到下标为j的矩阵需要的最小乘法次数。
  2. 先从2开始遍历连乘的矩阵的长度，再从最左边开始遍历最左边的矩阵，接着从最左边开始遍历最后一个乘号的位置。
  3. 最后返回dp[1][n]。

关键代码

int solve() {
    int n;
    cin >> n;

    for (int i = 1; i <= n; ++i) {
        cin >> r[i] >> c[i];
    }

    // dp[i][j]：从下标为i的矩阵连乘到下标为j的矩阵需要的最小乘法次数
    vector<vector<int>> dp(n+1,vector<int>(n+1,0));

    // 从2开始遍历连乘的矩阵的长度
    for (int len = 2; len <= n; len++) {
        for (int i = 1; i + len -1 <= n; i++) {
            int j = i + len -1;
            int min_cost = INT_MAX;
            
            // 遍历最后一个乘号的位置，k代表最后运算的乘号左边那个矩阵的下标
            for (int k = i; k <= j - 1; ++k) {
                int cost = dp[i][k] + dp[k + 1][j] + r[i] * c[k] * c[j];
                if (cost < min_cost) {
                    min_cost = cost;
                }
            }

            dp[i][j] = min_cost;
        }
    }
    return dp[1][n];
}

1021 钢条切割

题目

给定一根长度为n(n<=10000)的钢条以及一张价格表, 请计算这根钢条能卖出的最大总收益。价格表表示为(li,pi), 1<=i<=k. 不在价格表中的钢条可卖出价格为0。

思路

  1. 自定义结构体value存储价格表中长度为l的钢条对应的价格p。
  2. dp[i]：长度为i的钢条的最大价格。
  3. 先将价格表写入dp中。
  4. 然后从i=1开始遍历长度为i的钢条，通过遍历切的最后一刀的位置（根据对称性只需要从0遍历到i/2即可）真正确定dp[i]的值。
  5. 最后返回dp[len]。

关键代码

// 自定义钢条价值结构体，用来存储长度l对应的价值p
struct value {
    int l,p;
};

// 长度为len的钢条在价格表a下的最大收益
int f(vector<value>& a, int len) {
    // dp[i]：长度为i的钢条的最大价格
    vector<int> dp(len+1,0);
    
    for(int i=0;i<a.size();i++) {
        if(a[i].l>len) continue;
        dp[a[i].l]=a[i].p;
    }
    
    // 遍历钢条长度i
    for(int i=1;i<=len;i++) {
        // 遍历切的最后一刀的位置j
        for(int j=0;j<=i/2;j++) {
            dp[i]=max(dp[j]+dp[i-j],dp[i]);
        }
    }
    return dp[len];
}

1022 最长公共子序列

题目

给定两个字符串A和B, 请计算这两个字符串的最长公共子序列长度。

思路

  1. dp[i][j]：字符串s1的前i个字符与字符串s2的前j个字符的最长公共子序列长度。
  2. 逐行遍历dp数组，填入dp[i][j]。
  3. 最后返回dp[n][m]。

关键代码

// 返回s1与s2的最长公共子序列长度
int f(string& s1, string& s2) {
    int n=s1.size();
    int m=s2.size();
    
    // dp[i][j]：字符串s1的前i个字符与字符串s2的前j个字符的最长公共子序列长度
    vector<vector<int>> dp(n+1,vector<int>(m+1,0));
    
    for(int i=1;i<=n;i++) {
        for(int j=1;j<=m;j++) {
            if(s1[i-1]==s2[j-1]) dp[i][j]=dp[i-1][j-1]+1;
            else dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
        }
    }
    return dp[n][m];
}

1023 穷游？“穷”游？

题目

贫穷的小A有一个梦想，就是到t国去一次穷游，但现实是残酷的。小A所在的世界一共有n(n<=500)个国家，国家与国家之间总共有E(E<=50000)条道路相连，第i个国家对于进入它的外国人都要收取Bi的费用，而小A家住在s国，他必须通过这些道路在各个国家之间中转最终到达t国（除非他运气够好可以直接从s国到达t国）。但是贫穷的小A只剩下M(M<=100)元家底了，因此他必须精打细算旅途的
费用，同时小A对于t国实在太向往了，因此他希望能够走最短的路尽快到达t国。这个问题难倒了小A，现在他请你帮他算一算他到达t国的最短路径有多长。

第一行输入T(T<=10)表示有T组数据。每组数据第一行输入5个整数n、E、s、t、M，分别表示小A所在世界的国家数、国家之间的总道路数、小A的国籍、小A向往的国家以及小A的家底；接下来一行输入n个正整数Bi，表示第i个国家收取的过路费（由于小A是s国人，因此s国不会收取，但t国会）；接下来输入E行每行三个正整数u(1<=u<=n)、v(1<=v<=n)、w，表示u国和v国之间存在着一条长度为w的无向边（可能有重边）。输入保证最终结果不会使int溢出。

思路

  1. 自定义结构体Edge存储当前节点到另一个节点to的距离len。
  2. 自定义类Graph，利用Edge建立邻接表adj。
  3. 采用迪杰斯特拉算法的思想，dist[国家][花费] = 最短路程。
  4. 采用最小堆存储更新过距离的结点，循环取出最小堆顶部结点，尝试更新其周围结点到起点的最短距离，直到最小堆中没有元素为止。
  5. 返回dist[t][1~M]的最小值。

关键代码

const int INF = INT_MAX / 2;

// 自定义无向边结构体，存储当前节点到另一个节点to的距离len
struct Edge {
    int to, len;
    Edge(int t, int l) : to(t), len(l) {}
};

// 自定义地图类
class Graph {
public:
    vector<vector<Edge>> adj;

    Graph(int n) {
        adj.resize(n + 1);
    }

    void add(int u, int v, int w) {
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w);
    }
};

// 地图a中，各个国家收费表为c，共n个国家，小A在s国，想去t国，有M元钱
// 函数返回能走的最短路径长度
int solve(Graph& a, vector<int>& c, int n, int s, int t, int M) {
    // dist[国家][花费] = 最短路程
    vector<vector<int>> dist(n + 1, vector<int>(M + 1, INF));
    dist[s][0] = 0;

    // priority_queue< 元素类型, 底层容器, 比较规则 > 队列名;
    // pair比较规则：对于pair<first,second>，先比较first，若相等再比较second
    priority_queue<
        pair<int, pair<int, int>>,
        vector<pair<int, pair<int, int>>>,
        greater<pair<int, pair<int, int>>>> pq;
        // （路径长度，（国家，花费金额））

    pq.push({0, {s, 0}});

    while (!pq.empty()) {
        // 从最小堆中取出距离起点路径最短的点
        auto top = pq.top();
        pq.pop();
        int d = top.first;
        int u = top.second.first;
        int cost = top.second.second;

        // 若取出的这个点与起点的距离大于dist中存的距离就跳过
        if (d > dist[u][cost]) continue;

        // 遍历与该点相邻的点，试图更新起点到其他点的最短距离
        for (Edge e : a.adj[u]) {
            int v = e.to;
            int nd = d + e.len;
            int nc = cost;

            // 不是起点就收费
            if (v != s) {
                nc += c[v - 1];
            }

            // 累积金额超过限度就跳过
            if (nc > M) continue;
            // 新长度小于dist中存的距离就更新，并把更新的点压入最小堆中
            if (nd < dist[v][nc]) {
                dist[v][nc] = nd;
                pq.push({nd, {v, nc}});
            }
        }
    }

    int ans = INF;
    // 遍历所有金额下最短路径
    for (int co = 0; co <= M; ++co) {
        ans = min(ans, dist[t][co]);
    }
    return ans == INF ? -1 : ans;
}

1024 最优搜索二叉树

题目

给定N个整数关键字, 每个关键字有一搜索概率, 关键字外区间(共N+1个区间)也有搜索概率。可根据关键字构造二叉搜索树来减少搜索代价。假定二叉搜索树中关键字节点的搜索代价为该节点到树的根的路径上关键字节点的个数, 关键字之外区间的搜索代价也为该区间到树的根的路径上关键字节点的个数。二叉搜索树的期望代价为所有关键字和关键字之外区间的期望代价之和。请构造一棵最优二叉搜索树, 计算最优二叉搜索树的期望代价。

思路

  1. 先计算关键字概率前缀和，A[i] = a[0]+a[1]+...+a[i]。
  2. 再计算区间概率前缀和，C[i] = c[0]+c[1]+...+c[i]。
  3. dp[i][j]：表示由第 i ~ j 个关键字构成的子树对应的最小期望代价。
  4. 四边形不等式优化：根k的范围 [x[i][j-1], x[i+1][j]]。
  5. 返回dp[0][n-1]。

关键代码

// a是关键字概率序列，c是区间概率序列
double f(vector<double>& a, vector<double>& c, int n) {
    // dp[i][j]：表示由第 i ~ j 个关键字构成的子树，对应的最小期望代价
    vector<vector<double>> dp(n,vector<double>(n,DBL_MAX));

    // x[i][j]：记录区间 [i,j] 构成的子树的最优根节点下标
    vector<vector<int>> x(n,vector<int>(n,-1));

    // A数组：关键字概率前缀和，A[i] = a[0]+a[1]+...+a[i]
    vector<double> A(n);
    // C数组：区间概率前缀和，C[i] = c[0]+c[1]+...+c[i]
    vector<double> C(n+1);

    for(int i=0;i<n;i++) {
        // 单个节点i作为根：代价 = 当前关键字概率 + 左右两个相邻区间概率
        dp[i][i] = a[i] + c[i] + c[i+1];
        // 单个关键字构成的子树，根节点就是自身
        x[i][i] = i;

        // 计算前缀和 A、C
        if(i == 0) {
            A[0] = a[0];
            C[0] = c[0];
        }
        else {
            A[i] = A[i-1] + a[i];
            C[i] = C[i-1] + c[i];
        }
    }
    // 补充最后一个区间的前缀和 c[n]
    C[n] = C[n-1] + c[n];

    // s = j-i ，表示区间跨度，s从1开始（区间长度 >=2）
    for(int s=1;s<n;s++) {
        // 枚举区间左端点 i
        for(int i=0;i<n-s;i++) {
            // 由左端点+跨度得到区间右端点 j，当前处理区间 [i, j]
            int j = s + i;

            // 枚举区间 [i,j] 的根节点 k
            // 四边形不等式优化：根k的范围 [x[i][j-1], x[i+1][j]]
            // 不用暴力枚举 i~j，大幅减少计算量
            for(int k = x[i][j-1]; k <= x[i+1][j]; k++) {
                // 左子树代价：区间 [i, k-1]，若左区间不存在则代价为0
                double l = (k-1 >= i) ? dp[i][k-1] : 0.0;
                // 右子树代价：区间 [k+1, j]，若右区间不存在则代价为0
                double r = (k+1 <= j) ? dp[k+1][j] : 0.0;

                // 计算区间 [i,j] 内 所有关键字+区间 的总概率w
                // 总概率 = 关键字a[i..j] 和 + 区间c[i..j+1] 和
                double w;
                if(i > 0)
                    w = (A[j] - A[i-1]) + (C[j+1] - C[i-1]);
                else
                    w = A[j] + C[j+1];

                // 总期望代价 = 左子树代价 + 右子树代价 + 当前层总概率w
                // 最优树性质：以k为根时，所有节点/区间深度+1，等价于累加一次总概率w
                double t = l + r + w;

                // 找到更小的代价，更新dp值和最优根节点
                if(t < dp[i][j]) {
                    dp[i][j] = t;
                    x[i][j] = k;
                }
            }
        }
    }

    return dp[0][n-1];
}

1026 插入乘号

题目

给出N个1-9的数字 (v1,v2,…,vN), 不改变它们的相对位置, 在中间加入K个乘号和N-K-1个加号, 括号随便加, 使最终结果最大。因为乘号和加号一共就是N-1个，所以恰好每两个相邻数字之间都有一个符号。
例如： N=5, K=2，5个数字分别为1、2、3、4、5，可以进行如下运算：
1*2*(3+4+5)=24
1*(2+3)*(4+5)=45
(1*2+3)*(4+5)=45
等等。

思路

  1. 前缀和 sum[i] = a[0]+...+a[i]。
  2. dp[i][k%2]：前i个数字，插入k个乘号的最大值。因为dp递推公式中k这一项仅和k-1项有关，故可在这个维度上通过取模建立循环数组以达到节省空间的目的。
  3. 先从1开始遍历乘号数量k，再从k开始遍历dp[i][k%2]，然后遍历最右边一个乘号的位置。
  4. 最后返回dp[n-1][K%2]。

关键代码

// 计算n个数字插入K个乘号后运算结果的最大值
long long f(int n, int K, vector<int>& a) {
    // 前缀和 sum[i] = a[0]+...+a[i]
    vector<long long> sum(n, 0);
    sum[0]=a[0];
    for (int i = 1; i < n; i++) {
        sum[i] = a[i] + sum[i-1];
    }

    // dp[i][k%2] = 前 i 个数字，k 个乘号最大值
    vector<vector<long long>> dp(n, vector<long long>(2, 0));

    for (int i = 0; i < n; i++)
        dp[i][0] = sum[0][i];

    for (int k = 1; k <= K; k++) {
        for (int i = k; i < n; i++) {
            for (int j = k-1; j < i; j++) {
                // 最右边乘号后的数字全部加起来乘以前面的部分才能最大
                // 因为：A*(B*(s+...+t)+(S+...+T))<=A*(B*(s+...+t+S+...+T))
                dp[i][k%2] = max(dp[i][k%2], dp[j][(k-1)%2] * (sum[i]-sum[j]));
            }
        }
    }

    return dp[n-1][K%2];
}

1027 带权活动选择

题目

给定n个活动，活动ai表示为一个三元组(si,fi,vi)，其中si表示活动开始时间，fi表示活动的结束时间，vi表示活动的权重,si<fi。带权活动选择问题是选择一些活动，使得任意被选择的两个活动ai和aj执行时间互不相交，即区间[si,fi)与[sj,fj)互不重叠，并且被选择的活动的权重和最大。请设计一种方法求解带权活动选择问题。

思路

  1. 自定义结构体activity存储活动开始时间si，活动的结束时间fi，活动的权重vi。
  2. 从按活动结束时间从早到晚排好序的a中，采用二分法找到一个结束时间尽可能晚，但在下标为idx的活动开始之前的活动的下标，封装成函数find()。
  3. 对a[i]调用find()，存入p[i]中。
  4. dp[i-1]：前i个活动最大权值和。

关键代码

// 自定义活动结构体，存储活动开始时间si，活动的结束时间fi，活动的权重vi
struct activity {
    int s,f,v;
    bool operator<(activity& other) const {
        return f < other.f;
    }
};

// 从按活动结束时间从早到晚排好序的a中，采用二分法，
// 找到一个结束时间尽可能晚，但在下标为idx的活动开始之前的活动的下标
int find(vector<activity>& a, int idx) {
    int l=0;
    int r=idx-1;
    int t=a[idx].s;
    int res=-1;
    while(l<=r) {
        int mid=(l+r)/2;
        if(a[mid].f<=t) {
            res=mid;
            l=mid+1;
        }
        else {
            r=mid-1;
        }
    }
    return res;
}

// a中有n个活动，函数返回能够选择的活动的最大权值和
int f(vector<activity>& a, int& n) {
    // 按活动结束时间从早到晚对a进行排序
    sort(a.begin(),a.end());
    
    // 用数组p存储每一个活动执行find函数后得到的结果
    vector<int> p(n);
    for(int i=0;i<n;i++) {
        p[i]=find(a,i);
    }
    
    // dp[i-1]：前i个活动最大权值和
    vector<int> dp(n);
    dp[0]=a[0].v;
    for(int i=1;i<n;i++) {
        int t;
        if(p[i]!=-1) t=dp[p[i]]+a[i].v;
        else t=a[i].v;
        dp[i]=max(t,dp[i-1]);
    }
    return dp[n-1];
}

1028 迪杰斯特拉算法

题目

给定n(n<=500)个顶点,以及E(E<=10000)条边，使用迪杰斯特拉算法计算顶点s到顶点t的最短路径。

思路

  1. 自定义邻接表边结构体edge，存储目标节点、边权、下一条边指针。
  2. 自定义图类，采用邻接表存储无向图。
  3. 设置距离数组dist，起点距离置 0，其余节点初始为无穷大；设置标记数组vis，记录节点是否已确定最短路径。
  4. 每一轮从未确定最短路径的节点中，选出当前距离起点最近的节点，认定该节点的最短路径已经确定，打上标记。
  5. 遍历该节点的所有邻接节点，判断「起点→当前节点→邻接节点」的路径是否更短，若更短则更新邻接节点的距离。
  6. 重复上述选点、更新操作，直到找到目标终点，或所有节点遍历完毕。若最终终点距离仍为无穷大，说明两点无法连通，返回 - 1。

关键代码

// 邻接表边结构体：存储目标节点、边权、下一条边指针
struct edge {
    int v;       // 邻接节点编号
    int w;       // 边的权重
    edge* next;  // 指向下一条边的指针

    edge(int v_, int w_, edge* next_) : v(v_), w(w_), next(next_) {}
};

// 图类，采用邻接表存储无向图
class Graph {
public:
    // 构造函数：初始化n个节点的图，创建邻接表头指针数组
    Graph(int x):n(x) {
        p = new edge*[n+1](); // 节点从1开始编号，数组大小n+1
    }

    ~Graph() {
        for (int i = 1; i <= n; i++) {
            edge* now = p[i];
            // 遍历当前节点的所有边，逐个释放
            while (now != nullptr) {
                edge* tmp = now;
                now = now->next;
                delete tmp;
            }
        }
        delete[] p;
    }

    // 添加无向边：u <-> v，权重w
    void add(int u, int v, int w) {
        // 头插法：u指向v
        p[u] = new edge(v, w, p[u]);
        // 头插法：v指向u（无向图双向添加）
        p[v] = new edge(u, w, p[v]);
    }

    int n;         // 图中节点总数
    edge** p;      // 邻接表头指针数组
};

// 图a中有n个节点，s为起点，t为终点，函数返回s到t最短距离，不可到达返回-1
int dij(int n, int s, int t, Graph& a) {
    vector<bool> vis(n+1, 0);  // 标记数组：vis[j]=1 表示节点j已确定最短路径
    vector<int> dist(n+1, MAX);// 距离数组：dist[j] 表示起点到j的当前最短距离
    dist[s] = 0;

    // 最多遍历n轮，每轮确定一个节点的最短路径
    for(int i = 1; i <= n; i++) {
        int u = -1;
        int min_dist = MAX;

        // 1. 遍历所有未确定最短路径的节点，找到距离起点最近的节点u
        for(int j = 1; j <= n; j++) {
            if(!vis[j] && dist[j] < min_dist) {
                min_dist = dist[j];
                u = j;
            }
        }

        // 没有可访问的节点，说明终点不可达
        if(u == -1) return -1;
        // 提前找到终点，直接返回最短距离
        if(u == t) return dist[t];

        vis[u] = 1; // 标记节点u的最短路径已确定

        // 2. 遍历u的所有邻接边，松弛更新距离
        edge* v = a.p[u];
        while(v) {
            // 经过u到达邻接节点的路径长度
            int temp = dist[u] + v->w;
            // 若新路径更短，更新距离
            if(!vis[v->v] && temp < dist[v->v]) {
                dist[v->v] = temp;
            }
            v = v->next;
        }
    }
    return dist[t];
}

1029 最小生成树

题目

给定n(n<=500)个顶点,以及E(E<=20000)条边，计算最小生成树的权值。

第一行输入T表示有T组数据。每组数据第一行输入n、E，分别表示顶点数和边数。接下来输入E行每行三个正整数u(1<=u<=n)、v(1<=v<=n)、w，表示顶点u到顶点v之间无向边长度w（可能有重边）。

思路

  1. 自定义边结构体Edge：存储图的一条边，包含两个端点和权值。
  2. 自定义图类Graph：管理所有边。
  3. 自定义并查集DSU：用于判断连通性、合并集合，规避环。
  4. 优先选择权重最小的边，在不构成环的前提下，逐步将边加入生成树，最终选出n-1条边构成最小生成树。

关键代码

// 边结构体：存储图的一条边，包含两个端点和权值
struct Edge {
    int u, v, w;  // u、v为边的两个顶点，w为边的权重

    // 构造函数：初始化边的信息
    Edge(int u_, int v_, int w_) :u(u_), v(v_), w(w_) {}

    // 重载小于运算符：按边的权重升序排序，为克鲁斯卡尔算法做准备
    bool operator<(const Edge& other) const {
        return w < other.w;
    }
};

// 图类：管理所有边
class Graph {
public:
    vector<Edge> a;  // 存储图中所有的边

    Graph() {}

    ~Graph() {}

    // 添加一条边到图中
    void add(int u, int v, int w) {
        a.emplace_back(u, v, w);
    }

    // 对所有边按权重从小到大排序
    void init() {
        sort(a.begin(), a.end());
    }
};

// 并查集：用于判断连通性、合并集合，规避环
class DSU {
public:
    int* parent;  // 父节点数组，记录每个节点的祖先
    int* rank;    // 秩数组，用于按秩合并，优化树的高度
    int n;        // 顶点总数

    // 构造函数：初始化并查集，每个节点初始自成一个集合
    DSU(int N):n(N) {
        parent = new int[n + 1];
        rank = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            parent[i] = i;  // 每个节点的父节点初始为自身
            rank[i] = 0;    // 初始秩为0
        }
    }

    ~DSU() {
        delete[] parent;
        delete[] rank;
    }

    // 查找根节点 + 路径压缩优化
    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);  // 路径压缩：节点直接指向根节点
        }
        return parent[x];
    }

    // 判断两个节点是否在同一个连通集合中
    bool is_connected(int x, int y) {
        return find(x) == find(y);
    }

    // 合并两个集合（按秩合并），合并成功返回true，已连通返回false
    bool unite(int x, int y) {
        int root_x = find(x);
        int root_y = find(y);
        // 两节点根相同，属于同一集合，合并失败（会形成环）
        if (root_x == root_y) return false;

        // 小树合并到大树下，控制树高
        if (rank[root_x] < rank[root_y]) {
            parent[root_x] = root_y;
        }
        else {
            parent[root_y] = root_x;
            // 两棵树秩相同时，新根的秩+1
            if (rank[root_x] == rank[root_y]) {
                rank[root_x]++;
            }
        }
        return true;
    }
};

// 核心求解函数：克鲁斯卡尔最小生成树
void solve() {
    int n, E;
    cin >> n >> E;  // n:顶点数  E:边数
    Graph a;
    DSU dsu(n);     // 创建对应顶点数的并查集
    int u, v, w;

    // 读取所有边并加入图中
    for (int i = 0; i < E; i++) {
        cin >> u >> v >> w;
        a.add(u, v, w);
    }

    a.init();  // 将所有边按权重升序排序

    int count = 0;  // 已选中的生成树边数
    int total = 0;  // 最小生成树总权值

    // 遍历所有边，依次尝试加入生成树
    for (const auto& e : a.a) {
        // 两个顶点不连通，则选中这条边
        if (dsu.unite(e.u, e.v)) {
            count++;
            total += e.w;
            // 最小生成树共有 n-1 条边，凑齐直接输出结果
            if (count == n - 1) {
                cout << total << endl;
                return;
            }
        }
    }
    // 遍历完所有边仍不足 n-1 条，说明图不连通，无最小生成树
    cout << -1 << "\n";
}

1030 黑白连线

题目

给定直线上2n个点的序列P[1,2,… ,2n]，每个点P[i]要么是白点要么是黑点，其中共有n个白点和n个黑点，相邻两个点之间距离均为1，请设计一个算法将每个白点与一黑点相连，使得连线的总长度最小。例如，图中有4个白点和4个黑点，以图中方式相连，连线总长度为1+1+1+5=8。

第一行输入m表示有m组测试。每组测试首先输入n(n<=10000)，接下来输入2n个0或者1, 分别表示白色或者黑色, 其中0和1的个数分别为n个。

思路

  1. 贪心算法：每次让最左边的黑点与最左边的白点连接。

关键代码

void solve() {
    int n;
    cin>>n;
    int count=0;
    bool t;
    queue<int> b,w;	// 用队列记录黑点、白点位置
    int res=0;
    for(int i=0;i<2*n;i++) {
        cin>>t;
        if(t) {
            b.push(count++);
            if(!w.empty()) {
                while(!b.empty()&&!w.empty()) {
                    res+=abs(b.front()-w.front());
                    b.pop();
                    w.pop();
                }
            }
        }
        else {
            w.push(count++);
            if(!b.empty()) {
                while(!b.empty()&&!w.empty()) {
                    res+=abs(b.front()-w.front());
                    b.pop();
                    w.pop();
                }
            }
        }
    }
    cout<<res<<endl;
}

1035 生产安排

题目

某公司有个工厂和仓库。由于原材料等价格波动，工厂每个月的生产成本也会波动，令第𝑖个月产品的单位生产成本为𝑐_𝑖（该月生产一个产品的成本为𝑐_𝑖）。仓库储存产品的也有成本，假设每个月产品的单位储存成本为固定值1（存储一个产品一个月的成本为1）。令第𝑖个月需要供应给客户的产品数量为𝑦_𝑖，仓库里的和生产的产品均可供应给客户。假设仓库的容量无限大，供应给客户剩余的产品可储存在仓库中。若已知𝑛个月中各月的单位生产成本𝑐_𝑖、以及产品供应量𝑦_𝑖，设计一算法决策每个月的产品生产数量𝑥_𝑖，使得𝑛个月的总成本最低。例如：𝑛=3，𝑐_𝑖:2,5,3，𝑦_𝑖:2,4,5，则𝑥_𝑖:6,0,5，即第1个月生产6个供应2个（代价2×2=4），储存4个供应给第2个月（代价(2+1)×4=12），第3个月生产5个供应5个（代价3×5=15），使总成本4+12+15=31最小。

思路

  1. 维护一个变量**minCost用来存储当前遍历到的这个月的最低单价**，minCost=min(minCost+1,c[i])。
  2. 每个月都用minCost的单价去供应客户。

关键代码

void solve() {
    int n;
    cin>>n;
    vector<int> c(n);
    int t,minCost;	// minCost用来存储当前遍历到的这个月的最低单价
    for(int i=0;i<n;i++) {
        cin>>c[i];
    }
    minCost=c[0]-1;
    long long res=0;
    
    // 从第一个月开始遍历
    for(int i=0;i<n;i++) {
        cin>>t;
        minCost=min(minCost+1,c[i]);	// 更新minCost的值
        res+=1LL*minCost*t;
    }
    cout<<res<<endl;
}

1039 拓扑排序

题目

输入一张有向图, 输出拓扑排序后的的结果。

思路

  1. 自定义图类Graph，用邻接表存储有向图。

关键代码

// 图类，实现拓扑排序功能（基于邻接表）
class Graph {
private:
    int n;                  // 图中节点总数（节点编号 1~n）
    vector<vector<int>> adj;// 邻接表，adj[u] 存储 u 指向的所有后继节点
    vector<int> in_degree;  // 入度数组，in_degree[v] 表示节点 v 的入度

public:
    // 构造函数：初始化节点数量、邻接表、入度数组
    Graph(int n_) :n(n_) {
        adj.resize(n+1);        // 节点从1开始编号，数组大小+1
        in_degree.assign(n+1,0);// 所有节点初始入度置为0
    }

    ~Graph() {}

    // 添加一条有向边 x -> y
    void Add(int x, int y) {
        adj[x].push_back(y);    // y 加入 x 的邻接列表
        in_degree[y]++;         // 节点 y 入度 +1
    }

    // 读取边信息，构建整张图
    void Init() {
        int M;
        cin>>M;                 // 输入边的总条数
        int x,y;
        while(M--) {
            cin>>x>>y;          // 依次读取每条边的起点、终点
            Add(x,y);           // 调用函数添加边
        }
    }

    // 将所有入度为0的节点加入队列
    void Zero_in(queue<int>& q) {
        q = queue<int>();       // 清空队列
        for(int i=1;i<=n;i++)   // 遍历所有节点
            if(in_degree[i]==0)
                q.push(i);      // 入度为0，入队
    }

    // 核心：拓扑排序主逻辑
    void Topo() {
        queue<int> q;
        vector<int> res;        // 存储最终拓扑排序结果
        Zero_in(q);             // 先把所有入度为0的节点入队

        // 循环处理队列，直到队列为空
        while(!q.empty()) {
            int u=q.front();    // 取出队首节点
            q.pop();
            res.push_back(u);   // 加入拓扑结果集

            // 遍历当前节点 u 的所有后继节点 v
            for(int v : adj[u]) {
                in_degree[v]--; // 删掉边 u->v，v 的入度减1
                // 若v入度变为0，说明其前驱全部处理完成，加入队列
                if(in_degree[v]==0) {
                    q.push(v);
                }
            }
        }

        // 结果集长度不等于节点数 → 图存在环，无拓扑序
        if(res.size()!=n) 
            cout<<0<<endl;
        else {
            // 正常输出拓扑排序序列
            for(int i=0;i<n;i++) {
                if(i) cout<<" ";
                cout<<res[i];
            }
            cout<<endl;
        }
    }
};

// 处理单组测试用例
void solve() {
    int N;
    cin>>N;             // 输入当前图的节点数
    Graph g(N);         // 创建图对象
    g.Init();           // 读入边，建图
    g.Topo();           // 执行拓扑排序并输出
}
